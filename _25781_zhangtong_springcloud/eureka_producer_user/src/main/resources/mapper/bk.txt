<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd" >
<mapper namespace="com.zt.eureka_producer_user.mapper.UserMapper" >

    <select id="selectUserByName" parameterType="User" resultType="User">
        SELECT *
        FROM user u
        <where>
            <if test='username != null || username != " "'>
                AND u.username LIKE CONCAT(CONCAT("%", #{username}), "%")
            </if>
        </where>
    </select>

    <select id="selectUserById" parameterType="User" resultType="User">
        SELECT *
        FROM user u
        <where>
            <if test='id != null || id != " "'>
                AND u.id = id
            </if>
        </where>
    </select>

    <insert id="createUser" parameterType="User">
        INSERT INTO
        user
        (username, birthday, sex, address)
        VALUES
        (#{username}, #{birthday}, #{sex}, #{address})
    </insert>

    <update id="updateUser" parameterType="User">
        UPDATE user u SET u.id = #{id}
        <where>
            <if test='username != null || username != " "'>
                AND u.username = username
            </if>
            <if test='birthday != null || birthday != " "'>
                AND u.birthday = birthday
            </if>
            <if test='sex != null || sex != " "'>
                AND u.sex = sex
            </if>
            <if test='address != null || address != " "'>
                AND u.address = address
            </if>
        </where>
    </update>

    <delete id="deleteUser" parameterType="User">
        DELETE FROM user u WHERE u.id = #{id}
    </delete>

</mapper>
